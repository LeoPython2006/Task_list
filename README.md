    статус(выполнена или нет)(bool)
    сроки(datetime)
    время создания(datetime)
    id пользователя(int)


2.Список
    id(int)
    название(str 60)
    id пользователя(int)

3.Пользователь
    id(int)
    имя (str 60)
    tg_id(int)
    время создания(datetime)
